# 序列化和反序列化二叉搜索树

# leetcode 449 (opens new window)

序列化是将数据结构或对象转换为一系列位的过程，以便它可以存储在文件或内存缓冲区中，或通过网络连接链路传输，以便稍后在同一个或另一个计算机环境中重建。

设计一个算法来序列化和反序列化 二叉搜索树 。 对序列化/反序列化算法的工作方式没有限制。 您只需确保二叉搜索树可以序列化为字符串，并且可以将该字符串反序列化为最初的二叉搜索树。

编码的字符串应尽可能紧凑。

示例：

输入：root = [2,1,3]
输出：[2,1,3]

理解：

简单前序遍历运用题。

/**
 * Definition for a binary tree node.
 * class TreeNode {
 *     public $val = null;
 *     public $left = null;
 *     public $right = null;
 *     function __construct($value) { $this->val = $value; }
 * }
 */

class Codec {
    function __construct() {
        
    }
  
    /**
     * @param TreeNode $root
     * @return String
     */
    function serialize($root) {
        if ($root == null) {
            return '#_';
        }
        $str = $root->val . '_';
        $str .= $this->serialize($root->left);
        $str .= $this->serialize($root->right);
        return $str;
    }

    /**
     * @param String $data
     * @return TreeNode
     */
    function deserialize($data) {
        $data = explode('_', $data);
        return $this->_deserialize($data);
    }
    
    function _deserialize(&$data) {
        $v = array_shift($data);
        if ($v == '#') {
            return null;
        }
        $root = new TreeNode($v);
        $root->left = $this->_deserialize($data);
        $root->right = $this->_deserialize($data);
        return $root;
    }
}

/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */

type Codec struct {
	vArr []string
}

func Constructor() Codec {
	return Codec{}
}

// Serializes a tree to a single string.
func (this *Codec) serialize(root *TreeNode) string {
	if root == nil {
		return "#_"
	}
	str := strconv.Itoa(root.Val) + "_"
	str += this.serialize(root.Left)
	str += this.serialize(root.Right)
	return str
}

// Deserializes your encoded data to tree.
func (this *Codec) deserialize(data string) *TreeNode {
	vArr := strings.Split(data, "_")
	this.vArr = vArr[:]
	return this._deserialize()
}

func (this *Codec) _deserialize() *TreeNode {
	v := this.vArr[0]
	this.vArr = this.vArr[1:]
	if v == "#" {
		return nil
	}
	root := &TreeNode{}
	root.Val, _ = strconv.Atoi(v)
	root.Left = this._deserialize()
	root.Right = this._deserialize()
	return root
}
